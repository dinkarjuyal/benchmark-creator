import torch, torch.nn as nn
class RMSNorm(nn.Module):
    def __init__(self, d, elementwise_affine=True, bias=False, eps=1e-6):
        super().__init__()
        self.eps = eps
        self.weight = nn.Parameter(torch.ones(d)) if elementwise_affine else None
        self.bias = nn.Parameter(torch.zeros(d)) if (elementwise_affine and bias) else None
    def forward(self, x):
        input_dtype = x.dtype
        var = x.to(torch.float32).pow(2).mean(-1, keepdim=True)
        x = x * torch.rsqrt(var + self.eps)
        if self.weight is None: return x.to(input_dtype)
        if self.weight.dtype in (torch.float16, torch.bfloat16): x = x.to(self.weight.dtype)
        x = x * self.weight
        if self.bias is not None: x = x + self.bias
        return x

x = torch.randn(2, 4, dtype=torch.bfloat16)
norm = RMSNorm(4, elementwise_affine=True)
norm.weight = nn.Parameter(norm.weight.to(torch.bfloat16))
out = norm(x)
print(out.dtype == torch.float32)
