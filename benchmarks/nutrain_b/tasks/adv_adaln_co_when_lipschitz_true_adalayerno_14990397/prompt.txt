import torch, torch.nn as nn
def rms_norm_no_affine(x, eps=1e-6):
    var = x.to(torch.float32).pow(2).mean(-1, keepdim=True)
    return (x * torch.rsqrt(var + eps)).to(x.dtype)
class AdaLayerNormContinuous(nn.Module):
    def __init__(self, d, cond_d, lipschitz=False):
        super().__init__()
        self.lipschitz = lipschitz
        self.norm = nn.LayerNorm(d, elementwise_affine=False)
        self.linear = nn.Linear(cond_d, d * 2, bias=not lipschitz)
    def forward(self, x, emb):
        scale, shift = self.linear(emb).chunk(2, dim=1)
        if self.lipschitz:
            scale = rms_norm_no_affine(scale); shift = rms_norm_no_affine(shift)
        return self.norm(x) * (1 + scale[:, None]) + shift[:, None]

x = torch.randn(1, 3, 16, dtype=torch.bfloat16)
emb = torch.randn(1, 8, dtype=torch.bfloat16)
layer = AdaLayerNormContinuous(16, 8, lipschitz=True).bfloat16()
_ = layer(x, emb)  # forward pass to initialize weights in bfloat16
# Now manually override linear layer weights to be all zeros
layer.linear.weight.data.zero_()
out_rms = torch.sqrt(torch.mean(layer.linear(emb).chunk(2,1)[0].float()**2)).item()
print(f"RMS of scale: {out_rms:.4f}")
